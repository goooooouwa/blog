# update apt source-list
sudo apt-get update

# install ssh server
sudo apt-get install openssh-server

# install curl
sudo apt-get install curl

# install rvm
\curl -sSL https://get.rvm.io | bash -s stable --ruby


# install git
sudo apt-get install git

# install zsh

# install oh-my-zsh

chsh -s /bin/zsh

make install
